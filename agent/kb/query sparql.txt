SELECT ?AREAGEOGRAFICA ?REGIONE ?PROVINCIA
WHERE { GRAPH ?G {
	?CODICESCUOLA miur:AREAGEOGRAFICA ?AREAGEOGRAFICA.
	?CODICESCUOLA miur:REGIONE ?REGIONE.
	?CODICESCUOLA miur:PROVINCIA ?PROVINCIA.
        ?CODICESCUOLA miur:CODICESCUOLA "GERC004014".
} }
GROUP BY ?AREAGEOGRAFICA ?REGIONE ?PROVINCIA
ORDER BY ?AREAGEOGRAFICA ?REGIONE ?PROVINCIA
LIMIT 50
